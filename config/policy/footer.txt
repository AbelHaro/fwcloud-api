
# Verify that we have all the needed commands.
check_cmds

ACTION="$1"
test -z "$ACTION" && ACTION="start"

case "$ACTION" in
  start)
    greeting_msg
    policy_apply
    routing_apply
    ;;

  stop)
    iptables_default_filter_policy ACCEPT
    policy_empty
    ;;

  restart|reload)
    $0 stop
    $0 start
    ;;

  block)
    policy_empty
    iptables_default_filter_policy DROP
    ;;

  policy)
    greeting_msg
    policy_apply
    ;;

  routing)
    greeting_msg
    routing_apply
    ;;

  status)
    status
    ;;

  install)
    chmod 700 "$0"

    if [ -f "/etc/issue" ]; then
      if [ "`cat /etc/issue | grep EdgeOS`" ]; then
        chown root:vyattacfg "$0"
        INSTALL_DIR="/config/scripts/post-config.d"
      else
        chown root:root "$0"
        INSTALL_DIR="/etc/fwcloud"
      fi
    else
      chown root:root "$0"
      INSTALL_DIR="/etc/fwcloud"
    fi

    test ! -d "$INSTALL_DIR" && {
      mkdir -m 700 -p "$INSTALL_DIR"
      chown root:root "$INSTALL_DIR"
    }

    if [ -f "/etc/openwrt_version" ]; then
      # Change the Unix Shell to /bin/sh because OpenWRT doesn't has bash.
      sed "s/^\#\!\/bin\/sh/\#\!\/bin\/sh \/etc\/rc.common/g" "$0" > "$INSTALL_DIR/$0"
      chmod 700 "$INSTALL_DIR/$0"
      rm -f "$0"

      if [ ! -f "/etc/rc.d/S19fwcloud" -a -d "/etc/rc.d" ]; then
        cd /etc/rc.d
        ln -s ../fwcloud/fwcloud.sh S19fwcloud
      fi
    else
      mv "$0" "$INSTALL_DIR"
    fi

    if [ "$SYSTEMCTL" -a ! -f "$SYSTEMD_FILE" ]; then
      create_systemd_service
      $SYSTEMCTL enable fwcloud
    fi

    if [ "$SYSTEMCTL" -a ! -f "$SYSTEMD_FILE_ROUTING" ]; then
      create_systemd_service_routing
      $SYSTEMCTL enable fwcloud-routing
    fi
    ;;

  *)
    echo "Usage $0 [start|stop|restart|reload|block|status|install|routing]"
    ;;
esac

exit 0
